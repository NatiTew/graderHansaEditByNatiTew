import java.util.Scanner;

public class Stack2 {
	public static void main(String[] args) {
		Scanner kb = new Scanner(System.in);
		StackX<String> web = new StackX<String>();
		StackX<String> web2 = new StackX<String>();

		while (true) {
			String key = kb.next();
			if (key.equals("open")) {
				System.out.println("my homepage");
			} else if (key.equals("back")) {
				if (web.isEmpty() == true) {
					System.out.println("my homepage");
				} else {
					web.pop();
					if (web.peek() == null) {
						System.out.println("my homepage");
					} else {
						System.out.println(web.peek());
					}
				}
			} else if (key.equals("input")) {
				String key2 = kb.next();
				web.push(key2);
				System.out.println(key2);
			} else if (key.equals("close")) {
				System.exit(0);
			}
		}

	}
}
