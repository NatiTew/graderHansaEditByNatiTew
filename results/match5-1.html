import java.util.Scanner;

public class Main {
	public static void main(String[] args) {
		Scanner kb = new Scanner(System.in);
		int n = kb.nextInt();

		LinkList StudInfo = new LinkList();

		for (int i = 0; i < n; i++) {

			int m = kb.nextInt();
			for (int j = 0; j < m; j++) {
				String name = kb.next();
				double height = kb.nextDouble();
				String unit = kb.next();
				switch (unit) {

				case "m":
					StudInfo.insertStudInfo(name, height);break;
				case "dm":
					StudInfo.insertStudInfo(name, height / 10.000000);break;
				case "cm":
					StudInfo.insertStudInfo(name, height / 100.000000);break;
				case "mm":
					StudInfo.insertStudInfo(name, height / 1000.000000);break;
				}
			}System.out.println();
			StudInfo.display();
		}

	}

	private static class LinkList {
		private Link first;

		public LinkList() {
			first = null;
		}

		public void insertStudInfo(String name, double height) {
			Link newLink = new Link(name,height);
			if (first == null) {
				newLink.next = first;
				first = newLink;
			} else {
				
				Link current = first;
				
				if (height > current.sHeight) {
					newLink.next = current;

				} else if (height < current.sHeight) {
					newLink.next = current.next;
					current.next = newLink;
				} else {
					if (name.compareToIgnoreCase(current.sName) < 0) {
						newLink.next = current;
						current.next = newLink;
					} else {
						newLink.next = current.next;
						current.next = newLink;
					}
				}
				current = current.next;

				while (current.next != null) {
					if (height > current.sHeight) {
						newLink.next = current;

					} else if (height < current.sHeight) {
						newLink.next = current.next;
						current.next = newLink;
					} else {
						if (name.compareToIgnoreCase(current.sName) < 0) {
							newLink.next = current;
							current.next = newLink;
						} else {
							newLink.next = current.next;
							current.next = newLink;
						}
					}
					current = current.next;
				}
			}
		}
		public void display(){
			
			Link current = first;
			while(current!=null){
			current.display();
			current = current.next;
			}
			
			
		}
		// /////////////////////////////////////////////////////////////////
	}

	private static class Link {

		public String sName;
		public double sHeight;

		public Link next;

		public Link(String name, double height) {
			sName = name;
			sHeight = height;

		}

		public void display() {
			
			System.out.println(sName);
		}
	}
}
